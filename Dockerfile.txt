# 构建阶段
FROM node:18-alpine AS builder

WORKDIR /app

# 复制 package.json 和 package-lock.json
COPY package*.json ./

# 安装依赖
RUN npm install

# 复制项目所有文件
COPY . .

# 编译打包项目 (前端和后端)
RUN npm run build

# 运行阶段
FROM node:18-alpine

WORKDIR /app

# 只复制运行所需的文件
COPY --from=builder /app/package*.json ./
COPY --from=builder /app/dist ./dist

# 安装生产环境依赖
RUN npm install --omit=dev

# 暴露 3000 端口
EXPOSE 3000

# 启动服务器
CMD ["npm", "start"]
